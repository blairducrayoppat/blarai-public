### NEW (lines 446–463 — APPLIED):

```python
@mcp.tool()
def complete_task(task_id: int) -> str:
    """Mark a task as done.

    Reads current title, description, and priority before posting, so that
    Vikunja's full-object POST semantics do not zero absent fields (A33 fix).

    Args:
        task_id: The task ID to complete.
    """
    current = _api_get(f"/tasks/{task_id}")
    body = {
        "title": current["title"],
        "description": current.get("description") or "",
        "priority": current.get("priority", 0),
        "done": True,
    }
    task = _api_post(f"/tasks/{task_id}", body)
    return json.dumps({"id": task["id"], "title": task["title"], "done": True}, indent=2)
```
